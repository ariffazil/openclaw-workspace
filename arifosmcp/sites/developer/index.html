Python MCP client (async) 
from mcp import ClientSession
from mcp.client.streamable_http import streamablehttp_client
import asyncio

async def main():
    async with streamablehttp_client(
        "https://arifosmcp.arif-fazil.com/mcp"
    ) as (read, write, _):
        async with ClientSession(read, write) as session:
            await session.initialize()

            # Step 1: Initialize governed session
            anchor = await session.call_tool(
                "init_anchor_state",
                {"declared_name": "ResearchAgent"}
            )
            session_id = anchor.content[0].text

            # Step 2: Ground in real-world facts
            evidence = await session.call_tool(
                "reality_compass",
                {"query": "climate data 2026"}
            )

            # Step 3: Full governed reasoning kernel
            result = await session.call_tool(
                "arifOS_kernel",
                {
                    "query": "Analyze the evidence and provide a governed verdict",
                    "risk_tier": "medium",
                    "use_heart": True,
                    "session_id": session_id
                }
            )

            # Verdict: SEAL | HOLD | VOID
            print(result.content[0].text)

asyncio.run(main())
